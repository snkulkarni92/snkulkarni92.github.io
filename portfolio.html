Path Finding Algorithm (Artificial Intelligence Lab) 
/*	Path Finding for a 2D map with obstacles.
*	Enter the dimensions of the field/map and the source and destination co-ordinates.
*	Then enter the quantity of obstacles followed by their co-ordinates.
*	An optimal path will be displayed after input is entered correctly.
*	Algorithm used: A Star
*	Compiler used: gcc
*	
*	Saurabh Kulkarni
*	Pune Institute of Computer Technology
*	12/9/2013
*/
#include<stdio.h>
#include<math.h>
typedef struct point {
	int flag;	//0 - none		1 - obstacle	2 - travelled
	int list;	//0 - none		1 - open list	2 - closed list
	int f, g, h, px, py;
} point;
int main() {
	int i, j, k, x, y, minf, t;
	point field[30][30];
	int openx[200];
	int openy[200];
	int count = 0, m, n, srcx, srcy, destx, desty;
	printf("Enter dimensions of grid: ");
	scanf("%d%d",&m,&n);
	printf("Enter Source Co-ordinates: ");
	scanf("%d%d",&srcx,&srcy);
	printf("Enter Destination Co-ordinates: ");
	scanf("%d%d",&destx,&desty);
	for(i = 0; i < m; i++) {
		for(j = 0; j < n; j++) {
			field[i][j].flag = field[i][j].list = field[i][j].f = field[i][j].g = field[i][j].h = field[i][j].px = field[i][j].py = 0;
		}
	}
	printf("Enter number of obstacles: ");
	scanf("%d",&i);
	while(i>0) {
		printf("Enter oordinates: ");
		scanf("%d %d",&x,&y);
		field[x][y].flag = 1;
		i--;
	}
	openx[i] = openy[i] = 0;
	x = srcx;
	y = srcy;
	t = 0;
	field[x][y].list = 2;
	field[x][y].f = -1;
	while(1) {
		if((x==destx)&&(y==desty))
			t = 1;
		if(t == 1) {
			field[x][y].flag = -1;
			if((x==srcx)&&(y==srcy))
				break;
			i = field[x][y].px;
			y = field[x][y].py;
			x = i;
		}
		else {
			for(i = ((x>1)?x-1:0); i <= ((x < m-1)?x+1:m); i++) {
				for(j = ((y>1)?y-1:0); j <= ((y < n-1)?y+1:n); j++) {
					if((i==x)&&(j==y))
						continue;
					if(field[i][j].flag == 1)
						continue;
					if(field[i][j].list == 0) {
						openx[count] = i;		//Add to open list
						openy[count] = j;
						field[i][j].list = 1;
						count++;
						field[i][j].px = x;
						field[i][j].py = y;		//Set Parent
						field[i][j].g = (((i-x) * (j-y) == 0)?10:14) + field[x][y].g;	//Set g() value
						if(abs(i-destx) < abs(j-desty))
							field[i][j].h = (abs(i-destx)) * 14 + (abs(j-desty) - abs(i-destx)) * 10;
						else
							field[i][j].h = (abs(i-destx) - abs(j-desty)) * 10 + (abs(j-desty)) * 14;
						field[i][j].f = field[i][j].g + field[i][j].h;					//Set f() value
					}
					else if(field[i][j].list == 1) {
						k = (((i-x) * (j-y) == 0)?10:14) + field[x][y].g;
						if(field[i][j].g > k) {
							field[i][j].g = k;
							field[i][j].px = x;
							field[i][j].py = y;
							field[i][j].f = field[i][j].g + field[i][j].h;
						}
					}
				}
			}
			printf("\n");
			for(i=0;i < m;i++) {
				for(j=0;j < n;j++) {
					if(field[i][j].f == -1)
						printf(" **** ");
					else if(field[i][j].flag == 1)
						printf(" #### ");
					else
						printf("%2d+%2d ",field[i][j].g,field[i][j].h);
				}
				printf("\n");
			}
			printf("\n");
			minf = m * n * 10;
			for(k = count-1; k >=0; k--) {
				if(field[openx[k]][openy[k]].f < minf) {
					minf = field[openx[k]][openy[k]].f;
					x = openx[k];
					y = openy[k];
					i = k;
				}
			}
			field[x][y].list = 2;
			field[x][y].f = -1;
			count--;
			for(k = i; k < count; k++) {
				openx[k] = openx[k+1];
				openy[k] = openy[k+1];
			}
		}
	}
	for(j = 0;j < n; j++)
		printf("___");
	printf("\n");
	for(i = 0; i < m; i++) {
		printf("|");
		for(j = 0;j < n; j++) {
			if(field[i][j].flag == -1)
				printf("++|");
			else if(field[i][j].flag == 1)
				printf("##|");
			else if(field[i][j].list == 1)
				printf("..|");
			else if(field[i][j].list == 2)
				printf("..|");
			else
				printf("  |");
		}
		printf("\n");
		for(j = 0;j < n; j++)
			printf("___");
		printf("_\n");
	}
	return 0;
}
Two Player Connect-4 Game (Artificial Intelligence Lab) 
/*	2 player Connect Four game
*	The board is placed vertically and has dimensions 7 x 6.
*	Players 'White' and 'Black' drop pieces into the columns one by one.
*	First player to obtain four pieces in a row, column or a diagonal wins.
*	Valid Input: An integer between 1 and 7 to represent the move for the player.
*	Compiler Used: gcc
*	
*	Surabh Kulkarni,
*	Pune Institute of Computer Technology
*	26/9/2013
*/
#include<stdio.h>
void display(int board[6][7]) {
	int i,j,z=0,sum=0;
	for(j=0;j<7;j++)
		printf("__%d_",j+1);
	printf("\n");
	for(i=0;i<6;i++) {
		printf("|");
		for(j=0;j<7;j++) {
			if(board[i][j] == 1)
				printf(" O |");
			else if(board[i][j] == 2)
				printf(" * |");
			else
				printf("   |");
		}
		printf("\n");
		for(j=0;j<7;j++)
			printf("____");
		printf("\n");
	}
}
int calculate(int board[6][7], int count[7]) {
	int z=0,sum=0;
	int i,j;
	for(i=0;i<6;i++) {
		z=0;
		for(j=0;j<7;j++) {
			if(board[i][j]==1) {
				if(z==0)
					z=-1;
				else if(z<0)
					z--;
				else {
					if(z==3)
						sum += 16;
					else if(z==2)
						sum += 4;
					z=-1;
				}
			}
			else if(board[i][j]==2) {
				if(z==0)
					z=1;
				else if(z>=0)
					z++;
				else {
					if(z==-3)
						sum -= 16;
					else if(z==-2)
						sum -= 4;
					z=1;
				}
			}
			else {
				if(z==3)
					sum += 16;
				else if(z==2)
					sum += 4;
				else if(z==-3)
					sum -= 16;
				else if(z==-2)
					sum -= 4;
				z=0;
			}
		}
	}
	for(i=0;i<6;i++) {
		z=0;
		for(j=0;j<7;j++) {
			if(board[j][i]==1) {
				if(z==0)
					z=-1;
				else if(z<0)
					z--;
				else {
					if(z==3)
						sum += 16;
					else if(z==2)
						sum += 4;
					z=-1;
				}
			}
			else if(board[j][i]==2) {
				if(z==0)
					z=1;
				else if(z>=0)
					z++;
				else {
					if(z==-3)
						sum -= 16;
					else if(z==-2)
						sum -= 4;
					z=1;
				}
			}
			else
			{
				if(z==3)
					sum += 16;
				else if(z==2)
					sum += 4;
				else if(z==-3)
					sum -= 16;
				else if(z==-2)
					sum -= 4;
				z=0;
			}
		}
	}
	return sum;
}
int check(int board[6][7], int count[7], int move) {
	int i,j,x,y,z;
	y = 6-count[move];
	x = board[y][move];
	if(count[move]>3) {
		for(i=0;i<3;i++) {
			if(board[i][move]==x) {
				if(board[i+1][move]==x) {
					if(board[i+2][move]==x) {
						if(board[i+3][move]==x) {
							printf("%s wins!!!\n",((x>1)?"Black":"White"));
							return 1;
						}
					}
				}
			}
		}
	}
	for(i=0;i<4;i++) {
		if(board[y][i]==x) {
			if(board[y][i+1]==x) {
				if(board[y][i+2]==x) {
					if(board[y][i+3]==x) {
						printf("%s wins!!!\n",((x>1)?"Black":"White"));
						return 1;
					}
				}
			}
		}
	}
	z = 0;
	for(i=-3;i<3;i++) {
		if((move+i>=0)&&(move+i<7)&&(y+i>=0)&&(y+i<6)) {
			if(board[y+i][move+i]==x) {
				if(z==3) {
					printf("%s wins!!!\n",((x>1)?"Black":"White"));
					return 1;
				}
				z++;
			}
			else
				z=0;
		}
	}
	z=0;
	for(i=-3;i<3;i++) {
		if((move+i>=0)&&(move+i<7)&&(y-i>=0)&&(y-i<6)) {
			if(board[y-i][move+i]==x) {
				if(z==3) {
					printf("%s wins!!!\n",((x>1)?"Black":"White"));
					return 1;
				}
				z++;
			}
			else
				z=0;
		}
	}
	return 0;
}
int main() {
	int board[6][7];
	int count[7];
	int i,j,x,y;
	int turn=-1;							//even-white   odd-black
	for(i=0;i<6;i++)
		for(j=0;j<7;j++)
			board[i][j] = 0;
	for(i=0;i<7;i++)
		count[i] = 0;
	display(board);
	while(turn<41) {
		turn++;
		printf("\nEnter move for %s:",((turn%2>0)?"Black":"White"));
		scanf("%d",&x);
		while((count[x-1]==6)||(x<1)||(x>7)) {
			printf("Incorrect choice! Enter again: ");
			scanf("%d",&x);
		}
		x--;
		count[x]++;
		board[6-count[x]][x]=turn%2+1;
		display(board);
		printf("Strength: %d\n",calculate(board,count));
		if(check(board,count,x))
			return 0;
	}
	return 0;
}

Sniffnet
This project was done as per the fourth year syllabus for B.E. Computer Engineering.
Synopsis of the project
    Data analysis and report generation has become an integral part when it comes to business intelligence. To generate analytics, a web service provider needs to process data which is coming in and going out from the server. Before Cloud computing, capturing of data was done on premise using dedicated hardware. Post Cloud era, service providers are moving their servers in the Cloud. Though there is a reduction in maintenance cost, there is a lack of transparency. Web service providers need to modify the application code to incorporate capturing data and making sense out of it. One solution is to run network monitoring tools like tcpdump or WireShark. However, these tools use up a significant amount of memory thus leaving less memory available for critical user processes to run. A novel solution is proposed here for monitoring network activity of VMs which overcomes the disadvantages of previous solutions.
    The system uses Software Defined Networking (SDN) approach by utilizing Open vSwitch (OVS), an open-source virtual network switch. The switch is used to capture network flow and store it in a network-mounted file system (NFS). For fast storage and retrieval, data is stored in compressed form using LZO compression algorithm and indexed using bitmap indexing technique. Storing data in this way helps perform analysis and report generation for providing transparency to the Cloud tenants. The user interacts with the system through Software as a Service (SaaS) interface by firing queries specific to the data that is sought. Recorded network traffic can be viewed which facilitates its analysis.
